  as
  begin
    declare @cno varchar(20)
    declare @maxgrade int
    --定义游标
    declare mycursor cursor for select cid from course
    --打开游标
    open mycursor
    --取值
    fetch next from mycursor into @cno

    while (@@FETCH_STATUS=0)
